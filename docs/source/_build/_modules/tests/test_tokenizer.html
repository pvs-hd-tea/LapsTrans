    assert result == ['def', 'increment', '(', 'inp', ')', ':', 'NEWLINE_SYMBOL', 'TAB_SYMBOL', 'return', '[',
                      'each', '+', '1', 'for', 'each', 'in', 'inp', ']', 'NEWLINE_SYMBOL']
